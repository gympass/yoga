# Datepicker

### Reference

The `<Datepicker />` component is a date picker that allows the user to select a single date or a range of dates.

`startDate`, `endDate`, `disablePastFrom` and `disableFutureFrom` are expected to be in the UTC [`Date`](https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Date) format.

### Usage

### Single date Datepicker with locale

```javascript state
render(() => {
  const [selectedDate, setSelectedDate] = useState();

  return (
    <Datepicker
      type="single"
      startDate={selectedDate}
      onSelectSingle={setSelectedDate}
      locale={navigator.language}
    />
  );
});
```

### Range date Datepicker

```javascript state
render(() => {
  const [startDate, setStartDate] = useState();
  const [endDate, setEndDate] = useState();

  const onSelectRange = (startDate, endDate) => {
    console.log({ startDate, endDate });
  };

  return (
    <Datepicker
      type="range"
      startDate={startDate}
      endDate={endDate}
      onSelectRange={onSelectRange}
    />
  );
});
```

### Range date Datepicker with custom onSelectRange

```javascript state
render(() => {
  const [startDate, setStartDate] = useState();
  const [endDate, setEndDate] = useState();

  const onCustomSelectRange = selectedDate => {
    console.log({ selectedDate });
    if (!startDate || (startDate > selectedDate && !endDate) || endDate) {
      setStartDate(selectedDate);
      if (endDate) setEndDate(undefined);
    } else {
      setEndDate(selectedDate);
    }
  };

  return (
    <Datepicker
      type="range"
      startDate={startDate}
      endDate={endDate}
      customOnSelectRange={onCustomSelectRange}
    />
  );
});
```

### Single Datepicker with default selected date

```javascript state
render(() => {
  const d = new Date();
  const dateUTC = new Date(Date.UTC(d.getFullYear(), d.getMonth(), d.getDate()));
  d.setDate(dateUTC.getDate() + 2);
  const [selectedDate, setSelectedDate] = useState(dateUTC);

  return (
    <Datepicker
      type="single"
      startDate={selectedDate}
      onSelectSingle={setSelectedDate}
    />
  );
});
```

### Range Datepicker with default start date and end date

```javascript state
render(() => {
  const start = new Date();
  const end = new Date();
  const startUTC = new Date(Date.UTC(start.getFullYear(), start.getMonth(), start.getDate()));
  const endUTC = new Date(Date.UTC(end.getFullYear(), end.getMonth(), end.getDate()));

  endUTC.setDate(startUTC.getDate() + 15);
  const [startDate, setStartDate] = useState(startUTC);
  const [endDate, setEndDate] = useState(endUTC);

  const onSelectRange = (startDate, endDate) => {
    console.log({ startDate, endDate });
  };

  return (
    <Datepicker
      type="range"
      startDate={startDate}
      endDate={endDate}
      onSelectRange={onSelectRange}
    />
  );
});
```
